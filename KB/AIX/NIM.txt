MY_LPAR=lpar10
MY_POWER=Server-8202-E4C-SN06C85DR

chsysstate -m $MY_POWER -r lpar -n $MY_LPAR -o shutdown --immed --restart	# it will reboot an lpar immediately (without dump)
chsysstate -m $MY_POWER -r lpar -n $MY_LPAR -o shutdown --immed						#	will shutdown an LPAR immediatelly	

chsysstate -m $MY_POWER -r lpar -n $MY_LPAR -o on -b norm -f default			#will boot the LPAR normally
chsysstate -m $MY_POWER -r lpar -n $MY_LPAR -o on -b sms -f default				#will boot the LPAR in SMS mode

mkvterm -m $MY_POWER -p $MY_LPAR																					# opens terminal on my LPAR
rmvterm -m $MY_POWER -p $MY_LPAR																					# closes terminal on my LPAR

lssyscfg -m $MY_POWER -r lpar -F name,lpar_id,state,default_profile				#see LPAR's settings (important is default_profile to be used with -f above)
lssyscfg -r sys -F name:type_model:serial_num															# lists fields for every managed system
