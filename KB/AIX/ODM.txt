odmget -q "name=hdisk3" CuAt
odmget -q "name=rootvg" CuAt

lscfg		# device configuration
lsconf	# AIX configuration

lscfg -v		# details of all devices, manufacturer, type, model, part number
lsattr -El sys0	# provides attributes for name device 

synclvodm rootvg 		# reconstructs VGDA from ODM contained data
redefinevg rootvg 		# reconstructs ODM from VGDA contained data
chvg -g 				# like cfgmgr, rereads any PV changes, like size
